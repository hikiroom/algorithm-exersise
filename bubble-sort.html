const arr = [9, 5, 6, 10, 1, 3, 7, 2, 8, 4];
const bubbleSort = (arr) => {
    const newArr = [...arr];
    const arrLen = arr.length;

    for (let i = arrLen; i--;) {
        for (let j = arrLen; (arrLen - i) <= j; j--) {
            if (newArr[j] < newArr[j - 1]) {
                [newArr[j], newArr[j - 1]] = [newArr[j - 1], newArr[j]]
            }
        }
    }

    return newArr;
};
console.log(bubbleSort(arr));
