const arr = [9, 5, 6, 10, 1, 3, 7, 2, 8, 4];
const sleepSort = (arr) => {
    const newArr = [];
    const sleep = arr.map(number => new Promise(resolve => setTimeout(() => {
        newArr.push(number);
        resolve();
    }, number)));

    return Promise.all(sleep).then(() => newArr);
};
sleepSort(arr).then((newArr) => console.log(newArr));
