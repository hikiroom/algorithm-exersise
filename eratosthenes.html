const eratosthenes = (range) => {
    const result = Array(range).fill(true);
    result[0] = false;
    result[1] = false;

    for (let i = 0; i * i < range; i++) {
        if (result[i]) {
            for (let j = 2; i * j < range; j++) {
                result[i * j] = false;
            }
        }
    }

    return result;
};
console.log(eratosthenes(100));
