const MAX_ITERATION_NUMBER = 100;
const getPrimeNumber = (n, maxIterationNumber) => {
    let primeNumberOccurrences = 0;
    for (let i = 1; i <= maxIterationNumber; i++) {
        for (let j = 2; j <= i; j++) {
            if (i % j === 0 && j < i) {
                break;
            } else if (i % j === 0) {
                primeNumberOccurrences++;

                if (primeNumberOccurrences === n) {
                    return i;
                }
            }
        }
    }
};
console.log(getPrimeNumber(10, MAX_ITERATION_NUMBER));
