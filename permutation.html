const permutation = (n, r) => {
    const factorial = (baseNum) => {
        let result = baseNum;
        for (let i = result; --i; ) {
            result *= i;
        }
        return result;
    };

    return factorial(n) / factorial(n - r);
};
const permutationArr = (n, r) => {
    const result = [];
    const nArr = (() => {
        const arr = [];

        for (let i = 1; i <= n; i++) {
            arr.push(i);
        }

        return arr;
    })();

    for (let i = nArr.length; i--; ) {
        for (let j = nArr.length; j--; ) {
            const iItem = nArr[i];
            const jItem = nArr[j];

            if (iItem !== jItem) {
                result.push([nArr[i], nArr[j]]);
            }
        }
    }

    return result;
};
console.log(permutation(5, 2));
console.log(permutationArr(5, 2));
